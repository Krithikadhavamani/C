#include <stdio.h>
#include <math.h>
void main(){
    int a,b,c;
    float s,d,area;
    printf("Enter a three sides:");
    scanf("%d %d %d",&a,&b,&c);
    s=(a+b+c)/2;
    d=(s*(s-a)*(s-b)*(s-c));
    area=sqrt(d);
    printf("area of triangle %f square unit /n",area);
}
